```python
def calculate_tier_roi(opportunity_value: float, tier_cost: float, success_probability: float) -> float:

    expected_value = opportunity_value * success_probability
    roi = (expected_value - tier_cost) / tier_cost
```

```python
def optimize_tier_selection(opportunity_value: float, relationship_critical: bool, 
                          timeline_days: int, budget_remaining: float) -> str:
    """Algorithm for optimal tier selection"""
    
    # Calculate base ROI for each tier
    current_roi = (opportunity_value * 0.75 - 0.75) / 0.75
    standard_roi = (opportunity_value * 0.80 - 7.50) / 7.50  
    enhanced_roi = (opportunity_value * 0.85 - 22.00) / 22.00
    complete_roi = (opportunity_value * 0.90 - 42.00) / 42.00
    
    # Apply modifiers
    if relationship_critical:
        enhanced_roi *= 1.3  # 30% relationship bonus
        complete_roi *= 1.4  # 40% relationship bonus
        
    if timeline_days < 2:
        return "current"  # Time constraint override
        
    if budget_remaining < 7.50:
        return "current"  # Budget constraint override
        
    # Select tier with best ROI above threshold
    roi_threshold = 1000  # 1000% minimum ROI
    
    if complete_roi > roi_threshold and budget_remaining >= 42.00:
        return "complete"
    elif enhanced_roi > roi_threshold and budget_remaining >= 22.00:
        return "enhanced"  
    elif standard_roi > roi_threshold and budget_remaining >= 7.50:
        return "standard"
    else:
        return "current"
```

```python
class TierAppropriatenessValidator:
    """Validate tier selection appropriateness"""
    
    def validate_tier_selection(
        self, 
        tier: str, 
        opportunity: Dict, 
        profile: OrganizationProfile,
        budget: float
    ) -> ValidationResult:
        """Comprehensive tier appropriateness validation"""
        
        warnings = []
        recommendations = []
        
        opportunity_value = opportunity.get('estimated_value', 0)
        
        # Budget validation
        tier_cost = self.get_tier_cost(tier)
        if budget < tier_cost:
            warnings.append(f"Insufficient budget for {tier} tier")
            recommendations.append(f"Consider {self.recommend_tier_for_budget(budget)}")
            
        # Value validation  
        if tier == "complete" and opportunity_value < 500000:
            warnings.append("Complete tier may be over-analysis for this opportunity size")
            recommendations.append("Consider Enhanced tier for better cost-effectiveness")
            
        # Relationship validation
        if tier in ["enhanced", "complete"] and not self.requires_relationships(opportunity):
            warnings.append(f"{tier} tier includes relationship analysis that may not be needed")
            recommendations.append("Standard tier may provide sufficient analysis")
            
        return ValidationResult(
            is_appropriate=len(warnings) == 0,
            warnings=warnings,
            recommendations=recommendations
        )
```
